138class TestFileResolverIterator(unittest.TestCase):
139    """
140    TestFileResolverIterator exercises the io.FileResolverIterator class.
141    """
142
143    def test_iterator_from_dir(self):
144        """Iterate over a collection of FileResolvers (no predetermined order)"""
145
146        with tempfile.TemporaryDirectory() as temp_dir:
147            test_date = TEST_DATE
148            test_file_1 = create_test_file(temp_dir, test_date)
149            test_file_2 = create_test_file(temp_dir, test_date)
150            test_file_3 = create_test_file(temp_dir, test_date, prefix="unmatched")
151
152            with tempfile.TemporaryDirectory() as temp_dir_2:
153                test_file_4 = create_test_file(temp_dir_2, test_date)
154                test_file_5 = create_test_file(temp_dir_2, test_date)
155                test_file_6 = create_test_file(temp_dir_2, test_date, prefix="explicit")
156
157                iterator = io.FileResolverIterator(
158                    dir_name=temp_dir,
159                    radix_pattern=f"{FIXTURE_PREFIX}*.csv",
160                    input_files=[test_file_4, test_file_5, test_file_6],
161                    name_prefix=f"{FIXTURE_PREFIX}_",
162                )
163
164                resolved = []
165                for resolver in iterator:
166                    resolved.append(resolver.get_file_name())
167
168                self.assertEqual(len(resolved), 5)
169                self.assertNotIn(test_file_3, resolved)
170                for expected_file in [
171                    test_file_1,
172                    test_file_2,
173                    test_file_4,
174                    test_file_5,
175                    test_file_6,
176                ]:
177                    self.assertIn(expected_file, resolved)

180class TestFileResolverContext(unittest.TestCase):
181    """
182    TestFileResolverContext exercises the context manager for the io.FileResolver class.
183    """
184
185    def test_file_resolver_context_success(self):
186        """File should be moved to some predefined archive upon success"""
187
188        with tempfile.TemporaryDirectory() as temp_dir:
189            test_date = TEST_DATE
190            test_file = create_test_file(temp_dir, test_date)
191
192            file_resolver = io.FileResolver(file_name=test_file, archive_path="success")
193            with io.file_resolver_context(
194                file_resolver, runtime.Runtime(resources=None)
195            ):
196                # DO SOMETHING
197                pass
198
199            # file should be archived
200            self.assertFalse(pathlib.Path(file_resolver.get_file_name()).exists())
201            find = glob.glob(
202                str(pathlib.Path(temp_dir, "success", f"{FIXTURE_PREFIX}*.csv"))
203            )
204            found = list(find)
205            self.assertEqual(len(found), 1)
206
207    def test_file_resolver_context_failure(self):
208        """File should be moved to some predefined sink upon error"""
209
210        with tempfile.TemporaryDirectory() as temp_dir:
211            test_date = TEST_DATE
212            test_file = create_test_file(temp_dir, test_date)
213
214            file_resolver = io.FileResolver(file_name=test_file, bad_path="failed")
215
216            try:
217                with io.file_resolver_context(
218                    file_resolver, runtime.Runtime(resources=None)
219                ):
220                    # FAIL SOMEHOW, on specific (controlled) exception
221                    raise exceptions.InvalidFileError
222            except:  # that's ok here to mute exceptions, hence pylint: disable=bare-except
223                pass
224
225            # file should be moved to "failed"
226            self.assertFalse(pathlib.Path(file_resolver.get_file_name()).exists())
227            find = glob.glob(
228                str(pathlib.Path(temp_dir, "failed", f"{FIXTURE_PREFIX}*.csv"))
229            )
230            found = list(find)
231            self.assertEqual(len(found), 1)
232
233    def test_file_resolver_context_other_failure(self):
234        """File should NOT be moved on some uncontrolled error"""
235
236        with tempfile.TemporaryDirectory() as temp_dir:
237            test_date = TEST_DATE
238            test_file = create_test_file(temp_dir, test_date)
239
240            file_resolver = io.FileResolver(file_name=test_file, bad_path="failed")
241
242            try:
243                with io.file_resolver_context(
244                    file_resolver, runtime.Runtime(resources=None)
245                ):
246                    # FAIL SOMEHOW, on uncontrolled exception (e.g. bug, technical outage)
247                    raise TypeError
248            except:  # that's ok here to mute exceptions, hence pylint: disable=bare-except
249                pass
250
251            # file should NOT be moved
252            self.assertTrue(pathlib.Path(file_resolver.get_file_name()).exists())
